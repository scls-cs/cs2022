FPS = 30
count = 0

running = True
while running:
  clock.tick(FPS)
  for event in pygame.event.get():
    if event.type == pygame.QUIT:
       running = False
    if event.type == pygame.MOUSEBUTTONDOWN:
      count += 1
  if(count%2 == 0):
    screen.fill((255, 255, 255))
  else:
    screen.fill((0, 0, 0))
  pygame.display.flip()
